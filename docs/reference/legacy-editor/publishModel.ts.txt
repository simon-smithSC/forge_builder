export function validatePublishReadiness(
  course: PublishCourse,
  options: PublishReadinessOptions = {},
): PublishReadiness {
  const issues: PublishFinding[] = [];
  const warnings: PublishFinding[] = [];

  if (isBlank(course.title)) {
    issues.push({ code: "course_title_empty", message: "Course title is empty.", path: "course.title" });
  }

  if (course.lessons.length === 0) {
    issues.push({ code: "lessons_missing", message: "Course has no lessons.", path: "lessons" });
  }

  course.lessons.forEach((lesson, lessonIndex) => {
    if (isBlank(lesson.title)) {
      issues.push({
        code: "lesson_title_empty",
        message: "Lesson title is empty.",
        path: `lessons[${lessonIndex}].title`,
      });
    }

    if (lesson.type === "blocks" && (!lesson.blocks || lesson.blocks.length === 0)) {
      issues.push({
        code: "blocks_missing",
        message: "Lesson has no blocks.",
        path: `lessons[${lessonIndex}].blocks`,
      });
    }

    lesson.blocks?.forEach((block, blockIndex) => {
      if (isBlank(block.title)) {
        issues.push({
          code: "block_title_empty",
          message: "Block title is empty.",
          path: `lessons[${lessonIndex}].blocks[${blockIndex}].title`,
        });
      }

      if (isBlank(block.body)) {
        issues.push({
          code: "block_body_empty",
          message: "Block body is empty.",
          path: `lessons[${lessonIndex}].blocks[${blockIndex}].body`,
        });
      }

      if (block.image) {
        if (isBlank(block.image.src)) {
          issues.push({
            code: "image_src_missing",
            message: "Image block is missing a source URL.",
            path: `lessons[${lessonIndex}].blocks[${blockIndex}].image.src`,
          });
        }

        if (isBlank(block.image.alt)) {
          issues.push({
            code: "image_alt_missing",
            message: "Image block is missing alt text.",
            path: `lessons[${lessonIndex}].blocks[${blockIndex}].image.alt`,
          });
        }
      }

      validateBlockMedia(block, lessonIndex, blockIndex, issues, warnings);
      validateBlockInteractionData(block, lessonIndex, blockIndex, issues);
    });
  });

  if (!hasCompletionMarker(course)) {
    issues.push({
      code: "completion_marker_missing",
      message: "Course has no quiz, checklist, or interactive completion marker.",
      path: "lessons",
    });
  }

  if (options.apiState === "offline") {
    warnings.push({
      code: "api_offline",
      message: "API is offline; publish package will be local only.",
      path: "api",
    });
  }

  if (options.publishState === "draft") {
    warnings.push({
      code: "draft_unpublished",
      message: "Course is still an unpublished draft.",
      path: "course.state",
    });
  }

  return { ok: issues.length === 0, issues, warnings };
}

export function createPublishSnapshot(course: PublishCourse, settings: PublishSettings = {}): PublishSnapshot {
  const resolvedSettings: ResolvedPublishSettings = {
    launcher: settings.launcher ?? "index.html",
    tracking: settings.tracking ?? "local",
  };

  return {
    buildId: settings.buildId ?? createStableBuildId(course),
    title: course.title,
    createdAt: (settings.now ?? (() => new Date()))().toISOString(),
    settings: resolvedSettings,
    courseData: course,
    manifestFiles: createManifestFiles(course, resolvedSettings),
    statementProfile: {
      id: "forge.local.static-preview",
      verbs: ["initialized", "experienced", "completed", "answered"],
      activityTypes: ["course", "lesson", "block", "quiz"],
    },
  };
}

export function createXapiPackagePayload(snapshot: PublishSnapshot): DownloadPayload {
  return {
    filename: `${slugify(snapshot.title)}-${snapshot.buildId}.xapi.zip`,
    mime: "application/zip",
    content: createZipArchive(createXapiPackageFiles(snapshot)),
  };
}

export function estimatePackageSize(snapshot: PublishSnapshot): number {
  return createXapiPackagePayload(snapshot).content.byteLength;
}

function createManifestFiles(course: PublishCourse, settings: ResolvedPublishSettings): ManifestFile[] {
  return [
    { path: settings.launcher, role: "launcher", bytes: 2048 },
    { path: "course.json", role: "course-data", bytes: stableJsonByteLength(course) },
    { path: "manifest.json", role: "manifest", bytes: 512 },
    { path: "xapi/profile.json", role: "statement-profile", bytes: 768 },
    { path: "tincan.xml", role: "xapi-launch", bytes: 1024 },
  ];
}

type ZipSourceFile = {
  path: string;
  content: string;
};

function createXapiPackageFiles(snapshot: PublishSnapshot): ZipSourceFile[] {
  const courseJson = JSON.stringify(snapshot.courseData, null, 2);
  const profileJson = JSON.stringify(snapshot.statementProfile, null, 2);
  const manifestJson = JSON.stringify(
    {
      buildId: snapshot.buildId,
      title: snapshot.title,
      createdAt: snapshot.createdAt,
      packageType: "xapi",
      launcher: snapshot.settings.launcher,
      files: snapshot.manifestFiles,
      statementProfile: snapshot.statementProfile,
    },
    null,
    2,
  );

  return [
    { path: "index.html", content: createLauncherHtml(snapshot) },
    { path: "course.json", content: courseJson },
    { path: "manifest.json", content: manifestJson },
    { path: "xapi/profile.json", content: profileJson },
    { path: "tincan.xml", content: createTinCanXml(snapshot) },
  ];
}

              Array.from(answerList.children).forEach((child) => child.classList.remove("selected"));
            }
            if (selected.has(index)) {
              selected.delete(index);
              button.classList.remove("selected");
            } else {
              selected.add(index);
              button.classList.add("selected");
            }
          });
          answerList.append(button);
        });
        section.append(answerList, submitRow(() => {
          attempts += 1;
          const correct = setsEqual(selected, correctOptions(block));
          Array.from(answerList.children).forEach((child, index) => {
            child.classList.toggle("correct", correctOptions(block).has(index));
            child.classList.toggle("incorrect", selected.has(index) && !correctOptions(block).has(index));
          });
          recordAnswer(course, lesson, block, Array.from(selected).map((index) => options[index]).join("[,]"), correct, attempts);
          feedback.textContent = correct ? feedbackText(block, true) : feedbackText(block, false);
        }, () => selected.size > 0), feedback);
        return section;
      }

        });

      function markBlockComplete(course, lesson, block) {
        if (!completedBlocks.has(block.id)) {
          completedBlocks.add(block.id);
          sendStatement("http://adlnet.gov/expapi/verbs/completed", blockObject(course, lesson, block), { completion: true });
        }
        const blocks = lesson.blocks || [];
        if (blocks.length && blocks.every((item) => completedBlocks.has(item.id)) && !completedLessons.has(lesson.id)) {
          completedLessons.add(lesson.id);
          sendStatement("http://adlnet.gov/expapi/verbs/completed", lessonObject(course, lesson), { completion: true });
        }
        if (isCourseComplete(course)) {
          sendStatement("http://adlnet.gov/expapi/verbs/completed", courseObject(course), { completion: true, score: { scaled: 1 } });
        }
      }

function createTinCanXml(snapshot: PublishSnapshot): string {
  const courseId = courseActivityIri(snapshot.courseData);
  const lessonActivities = snapshot.courseData.lessons
    .map((lesson) => `    <activity id="${escapeXml(lessonActivityIri(snapshot.courseData, lesson))}" type="http://adlnet.gov/expapi/activities/module">
      <name>${escapeXml(lesson.title)}</name>
      <description>${escapeXml(`Lesson in ${snapshot.title}.`)}</description>
    </activity>`)
    .join("\n");
  const interactionActivities = snapshot.courseData.lessons
    .flatMap((lesson) => lesson.blocks ?? [])
    .filter(isInteractionBlock)
    .map((block) => createInteractionActivityXml(snapshot.courseData, block))
    .join("\n");

  return `<?xml version="1.0" encoding="utf-8"?>
<tincan xmlns="http://projecttincan.com/tincan.xsd">
  <activities>
    <activity id="${escapeXml(courseId)}" type="http://adlnet.gov/expapi/activities/course">
      <name>${escapeXml(snapshot.title)}</name>
      <description>Forge xAPI package generated ${escapeXml(snapshot.createdAt)}.</description>
      <launch lang="en-US">${escapeXml(snapshot.settings.launcher)}</launch>
    </activity>

  return `https://xapi.supercell.com/courses/${encodeURIComponent(course.id)}`;

  return `${courseActivityIri(course)}/lessons/${encodeURIComponent(lesson.id)}`;

  return `${courseActivityIri(course)}/interactions/${encodeURIComponent(block.id)}`;

function isInteractionBlock(block: PublishBlock): boolean {
  return block.family === "quiz" || block.family === "knowledgeCheck";
}

function createInteractionActivityXml(course: PublishCourse, block: PublishBlock): string {
  const choices = getQuizOptions(block)
    .map((option) => `      <choice id="${escapeXml(option)}"><description>${escapeXml(option)}</description></choice>`)
    .join("\n");
  const matchComponents = getQuizMatches(block)
    .map(
      (match) => `      <source id="${escapeXml(match.prompt)}"><description>${escapeXml(match.prompt)}</description></source>
      <target id="${escapeXml(match.match)}"><description>${escapeXml(match.match)}</description></target>`,
    )
    .join("\n");
  const correctResponses = getCorrectResponses(block)
    .map((response) => `      <correctResponsesPattern>${escapeXml(response)}</correctResponsesPattern>`)
    .join("\n");
  const components = [choices, matchComponents, correctResponses].filter(Boolean).join("\n");

  return `    <activity id="${escapeXml(interactionActivityIri(course, block))}" type="http://adlnet.gov/expapi/activities/cmi.interaction">
      <name>${escapeXml(getQuizQuestion(block))}</name>
      <interactionType>${escapeXml(getInteractionType(block))}</interactionType>
${components}
    </activity>`;
}

function getQuizQuestion(block: PublishBlock): string {
  const question = block.content?.question;
  return typeof question === "string" && question.trim() ? question : block.title;
}

function getInteractionType(block: PublishBlock): string {
  if (block.variant === "fill-blank") {
    return "fill-in";
  }
  if (block.variant === "matching") {
    return "matching";
  }
  return "choice";
}

function getQuizOptions(block: PublishBlock): string[] {
  const options = block.content?.options;
  return Array.isArray(options) ? options.map((option) => String(option)) : [];
}

function getQuizMatches(block: PublishBlock): Array<{ prompt: string; match: string }> {
  const matches = block.content?.matches;
  if (!Array.isArray(matches)) {
    return [];
  }
  return matches.map((match) => {
    if (match && typeof match === "object") {
      const record = match as Record<string, unknown>;
      return { prompt: String(record.prompt ?? ""), match: String(record.match ?? "") };
    }
    return { prompt: String(match), match: "" };
  });
}

function getCorrectResponses(block: PublishBlock): string[] {
  if (block.variant === "fill-blank") {
    const acceptedAnswers = block.content?.acceptedAnswers;
    if (!Array.isArray(acceptedAnswers)) {
      return [];
    }
    return acceptedAnswers.map((answer) => {
      if (answer && typeof answer === "object") {
        return String((answer as Record<string, unknown>).value ?? "");
      }
      return String(answer);
    });
  }

  if (block.variant === "matching") {
    return getQuizMatches(block).map((match) => `${match.prompt}[.]${match.match}`);
  }

  const options = getQuizOptions(block);
  const correctIndexes = new Set<number>();
  const correctOption = block.content?.correctOption;
  const correctOptions = block.content?.correctOptions;

  if (typeof correctOption === "number") {
    correctIndexes.add(correctOption);
  }
  if (Array.isArray(correctOptions)) {
    correctOptions.forEach((option) => correctIndexes.add(Number(option)));
  }

  return Array.from(correctIndexes)
    .map((index) => options[index])
    .filter((option): option is string => Boolean(option));
}

function createZipArchive(files: ZipSourceFile[]): Uint8Array {
  const encoder = new TextEncoder();
  const localParts: Uint8Array[] = [];
  const centralParts: Uint8Array[] = [];
  let offset = 0;

  files.forEach((file) => {
    const nameBytes = encoder.encode(file.path);
    const contentBytes = encoder.encode(file.content);
    const crc = crc32(contentBytes);
    const localHeader = createZipLocalHeader(nameBytes, contentBytes, crc);
    localParts.push(localHeader, contentBytes);
    centralParts.push(createZipCentralDirectoryHeader(nameBytes, contentBytes, crc, offset));
    offset += localHeader.byteLength + contentBytes.byteLength;
  });

  const centralDirectoryOffset = offset;
  const centralDirectorySize = centralParts.reduce((total, part) => total + part.byteLength, 0);
  const endRecord = createZipEndRecord(files.length, centralDirectorySize, centralDirectoryOffset);
  return concatBytes([...localParts, ...centralParts, endRecord]);
}

function createZipLocalHeader(nameBytes: Uint8Array, contentBytes: Uint8Array, crc: number): Uint8Array {
  const header = new Uint8Array(30 + nameBytes.byteLength);
  const view = new DataView(header.buffer);
  view.setUint32(0, 0x04034b50, true);
  view.setUint16(4, 20, true);
  view.setUint16(6, 0x0800, true);
  view.setUint16(8, 0, true);
  view.setUint16(10, 0, true);
  view.setUint16(12, 0, true);
  view.setUint32(14, crc, true);
  view.setUint32(18, contentBytes.byteLength, true);
  view.setUint32(22, contentBytes.byteLength, true);
  view.setUint16(26, nameBytes.byteLength, true);
  view.setUint16(28, 0, true);
  header.set(nameBytes, 30);
  return header;
}

function createZipCentralDirectoryHeader(nameBytes: Uint8Array, contentBytes: Uint8Array, crc: number, localHeaderOffset: number): Uint8Array {
  const header = new Uint8Array(46 + nameBytes.byteLength);
  const view = new DataView(header.buffer);
  view.setUint32(0, 0x02014b50, true);
  view.setUint16(4, 20, true);
  view.setUint16(6, 20, true);
  view.setUint16(8, 0x0800, true);
  view.setUint16(10, 0, true);
  view.setUint16(12, 0, true);
  view.setUint16(14, 0, true);
  view.setUint32(16, crc, true);
  view.setUint32(20, contentBytes.byteLength, true);
  view.setUint32(24, contentBytes.byteLength, true);
  view.setUint16(28, nameBytes.byteLength, true);
  view.setUint16(30, 0, true);
  view.setUint16(32, 0, true);
  view.setUint16(34, 0, true);
  view.setUint16(36, 0, true);
  view.setUint32(38, 0, true);
  view.setUint32(42, localHeaderOffset, true);
  header.set(nameBytes, 46);
  return header;
}

function createZipEndRecord(entryCount: number, centralDirectorySize: number, centralDirectoryOffset: number): Uint8Array {
  const record = new Uint8Array(22);
  const view = new DataView(record.buffer);
  view.setUint32(0, 0x06054b50, true);
  view.setUint16(4, 0, true);
  view.setUint16(6, 0, true);
  view.setUint16(8, entryCount, true);
  view.setUint16(10, entryCount, true);
  view.setUint32(12, centralDirectorySize, true);
  view.setUint32(16, centralDirectoryOffset, true);
  view.setUint16(20, 0, true);
  return record;
}

function concatBytes(parts: Uint8Array[]): Uint8Array {
  const totalLength = parts.reduce((total, part) => total + part.byteLength, 0);
  const combined = new Uint8Array(totalLength);
  let offset = 0;
  parts.forEach((part) => {
    combined.set(part, offset);
    offset += part.byteLength;
  });
  return combined;
}

function crc32(bytes: Uint8Array): number {
  let crc = 0xffffffff;
  bytes.forEach((byte) => {
    crc ^= byte;
    for (let bit = 0; bit < 8; bit += 1) {
      crc = (crc >>> 1) ^ (crc & 1 ? 0xedb88320 : 0);
    }
  });
  return (crc ^ 0xffffffff) >>> 0;
}

function hasCompletionMarker(course: PublishCourse): boolean {
  return course.lessons.some((lesson) => {
    if (lesson.type === "quiz") {
      return true;
    }

    return lesson.blocks?.some((block) => {
      const family = block.family.toLowerCase();
      const marker = block.completionMarker?.toLowerCase();
      return Boolean(block.completion) || completionFamilies.has(family) || Boolean(marker && completionFamilies.has(marker));
    });
  });
}

function validateBlockMedia(
  block: PublishBlock,
  lessonIndex: number,
  blockIndex: number,
  issues: PublishFinding[],
  warnings: PublishFinding[],
) {
  block.media?.forEach((asset, assetIndex) => {
    const path = `lessons[${lessonIndex}].blocks[${blockIndex}].media[${assetIndex}]`;
    if (asset.kind !== "code" && isBlank(asset.src)) {
      issues.push({
        code: "media_src_missing",
        message: `${asset.kind} asset is missing a source.`,
        path: `${path}.src`,
      });
    }

    if ((asset.kind === "image" || asset.kind === "gallery-image") && isBlank(asset.alt)) {
      warnings.push({
        code: "image_alt_missing",
        message: `${asset.kind} asset is missing alt text.`,
        path: `${path}.alt`,
      });
    }
  });
}

function validateBlockInteractionData(
  block: PublishBlock,
  lessonIndex: number,
  blockIndex: number,
  issues: PublishFinding[],
) {
  if (block.family === "buttons") {
    const buttons = objectArray(block.content?.buttons);
    buttons.forEach((button, buttonIndex) => {
      if (isBlank(stringValue(button.destination))) {
        issues.push({
          code: "button_destination_missing",
          message: "Button is missing a destination.",
          path: `lessons[${lessonIndex}].blocks[${blockIndex}].content.buttons[${buttonIndex}].destination`,
        });
      }
    });
  }

  if (block.family === "quiz" || block.family === "knowledgeCheck") {
    const variant = block.variant;
    const hasAnswers =
      Array.isArray(block.content?.correctOptions) && block.content.correctOptions.length > 0
        ? true
        : typeof block.content?.correctOption === "number"
          ? true
          : Array.isArray(block.content?.acceptedAnswers) && block.content.acceptedAnswers.length > 0
            ? true
            : Array.isArray(block.content?.matches) && block.content.matches.length > 0;

    if ((variant === "fill-blank" || variant === "matching" || variant === "multiple-response" || variant === "multiple-choice") && !hasAnswers) {
      issues.push({
        code: "quiz_answer_missing",
        message: "Quiz block is missing answer data.",
        path: `lessons[${lessonIndex}].blocks[${blockIndex}].content`,
      });
    }
  }
}

function objectArray(value: unknown): Array<Record<string, unknown>> {
  if (!Array.isArray(value)) {
    return [];
  }

  return value.map((item) => (item && typeof item === "object" ? { ...(item as Record<string, unknown>) } : { value: String(item) }));
}

function stringValue(value: unknown): string {
  return typeof value === "string" ? value : "";
}

function escapeHtml(value: string): string {
  return value.replace(/[&<>"']/g, (character) => {
    switch (character) {
      case "&":
        return "&amp;";
      case "<":
        return "&lt;";
      case ">":
        return "&gt;";
      case '"':
        return "&quot;";
      default:
        return "&#39;";
    }
  });
}

function escapeXml(value: string): string {
  return escapeHtml(value);
}

function isBlank(value: string | undefined): boolean {
  return !value || value.trim().length === 0;
}

function stableJsonByteLength(value: unknown): number {
  return new TextEncoder().encode(JSON.stringify(value, null, 2)).length;
}

function createStableBuildId(course: PublishCourse): string {
  const hash = Array.from(JSON.stringify(course)).reduce((accumulator, character) => {
    return (accumulator * 31 + character.charCodeAt(0)) >>> 0;
  }, 0);

  return `build_${hash.toString(36)}`;
}

function slugify(value: string): string {
  const slug = value
    .trim()
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-+|-+$/g, "");

  return slug || "course";
}
