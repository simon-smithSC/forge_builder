export type BlockFamily =
  | "text"
  | "heading"
  | "subheading"
  | "twoColumnText"
  | "statement"
  | "quote"
  | "list"
  | "image"
  | "gallery"
  | "divider"
  | "multimedia"
  | "interactive"
  | "process"
  | "flashcard"
  | "buttons"
  | "knowledgeCheck"
  | "chart"
  | "table"
  | "audio"
  | "scenario"
  | "callout"
  | "checklist"
  | "quiz";

export type MoveDirection = "up" | "down";

export type BlockContent = Record<string, unknown>;

export type BlockSettings = Record<string, unknown>;

export interface CourseBlock {
  id: string;
  family: BlockFamily;
  title: string;
  content: BlockContent;
  settings: BlockSettings;
  estimatedMinutes: number;
}

export interface Lesson {
  id: string;
  title: string;
  blocks: CourseBlock[];
}

export interface Course {
  id: string;
  title: string;
  lessons: Lesson[];
}

export interface CourseStats {
  lessons: number;
  blocks: number;
  quizzes: number;
  estimatedMinutes: number;
}

export interface BlockPatch {
  title?: string;
  content?: BlockContent;
  settings?: BlockSettings;
  estimatedMinutes?: number;
}

export const addLesson = (course: Course, title = "Untitled lesson"): Course => ({
  ...course,
  lessons: [
    ...course.lessons,
    {
      id: nextId(course, "lesson"),
      title,
      blocks: [],
    },
  ],
});

export const renameLesson = (course: Course, lessonId: string, title: string): Course =>
  updateLesson(course, lessonId, (lesson) => ({
    ...lesson,
    title,
  }));

export const deleteLesson = (course: Course, lessonId: string): Course => {
  if (course.lessons.length <= 1 || !course.lessons.some((lesson) => lesson.id === lessonId)) {
    return course;
  }

  return {
    ...course,
    lessons: course.lessons.filter((lesson) => lesson.id !== lessonId),
  };
};

export const moveLesson = (course: Course, lessonId: string, direction: MoveDirection): Course => {
  const lessonIndex = course.lessons.findIndex((lesson) => lesson.id === lessonId);
  const targetIndex = direction === "up" ? lessonIndex - 1 : lessonIndex + 1;

  if (lessonIndex === -1 || targetIndex < 0 || targetIndex >= course.lessons.length) {
    return course;
  }

  const lessons = [...course.lessons];
  const [lesson] = lessons.splice(lessonIndex, 1);

  if (!lesson) {
    return course;
  }

  lessons.splice(targetIndex, 0, lesson);

  return {
    ...course,
    lessons,
  };
};

export const addBlock = (
  course: Course,
  lessonId: string,
  family: BlockFamily,
  insertAtIndex?: number,
): Course =>
  updateLesson(course, lessonId, (lesson) => ({
    ...lesson,
    blocks: insertItem(lesson.blocks, clampInsertionIndex(insertAtIndex, lesson.blocks.length), {
      id: nextId(course, "block"),
      ...cloneBlockDefaults(family),
    }),
  }));

export const updateBlock = (
  course: Course,
  lessonId: string,
  blockId: string,
  patch: BlockPatch,
): Course =>
  updateBlockInLesson(course, lessonId, blockId, (block) => ({
    ...block,
    ...patch,
    content: patch.content ? { ...block.content, ...patch.content } : block.content,
    settings: patch.settings ? { ...block.settings, ...patch.settings } : block.settings,
  }));

export const deleteBlock = (course: Course, lessonId: string, blockId: string): Course =>
  updateLesson(course, lessonId, (lesson) => {
    if (!lesson.blocks.some((block) => block.id === blockId)) {
      return lesson;
    }

    return {
      ...lesson,
      blocks: lesson.blocks.filter((block) => block.id !== blockId),
    };
  });

export const duplicateBlock = (course: Course, lessonId: string, blockId: string): Course =>
  updateLesson(course, lessonId, (lesson) => {
    const blockIndex = lesson.blocks.findIndex((block) => block.id === blockId);

    if (blockIndex === -1) {
      return lesson;
    }

    const block = lesson.blocks[blockIndex];

    if (!block) {
      return lesson;
    }

    const duplicate: CourseBlock = {
      ...block,
      id: nextId(course, "block"),
      content: cloneRecord(block.content),
      settings: cloneRecord(block.settings),
    };
    const blocks = [...lesson.blocks];
    blocks.splice(blockIndex + 1, 0, duplicate);

    return {
      ...lesson,
      blocks,
    };
  });

export const moveBlock = (
  course: Course,
  lessonId: string,
  blockId: string,
  direction: MoveDirection,
): Course =>
  updateLesson(course, lessonId, (lesson) => {
    const blockIndex = lesson.blocks.findIndex((block) => block.id === blockId);
    const targetIndex = direction === "up" ? blockIndex - 1 : blockIndex + 1;

    if (blockIndex === -1 || targetIndex < 0 || targetIndex >= lesson.blocks.length) {
      return lesson;
    }

    const blocks = [...lesson.blocks];
    const [block] = blocks.splice(blockIndex, 1);
    if (!block) {
      return lesson;
    }

    blocks.splice(targetIndex, 0, block);

    return {
      ...lesson,
      blocks,
    };
  });

export const getCourseStats = (course: Course): CourseStats => {
  const blocks = course.lessons.flatMap((lesson) => lesson.blocks);

  return {
    lessons: course.lessons.length,
    blocks: blocks.length,
    quizzes: blocks.filter((block) => block.family === "quiz" || block.family === "knowledgeCheck").length,
    estimatedMinutes: blocks.reduce((total, block) => total + block.estimatedMinutes, 0),
  };
};

export const findBlock = (
  course: Course,
  lessonId: string,
  blockId: string,
): CourseBlock | undefined =>
  course.lessons.find((lesson) => lesson.id === lessonId)?.blocks.find((block) => block.id === blockId);

const updateLesson = (
  course: Course,
  lessonId: string,
  updater: (lesson: Lesson) => Lesson,
): Course => {
  const lessonIndex = course.lessons.findIndex((lesson) => lesson.id === lessonId);

  if (lessonIndex === -1) {
    return course;
  }

  const currentLesson = course.lessons[lessonIndex];

  if (!currentLesson) {
    return course;
  }

  const nextLesson = updater(currentLesson);

  if (nextLesson === currentLesson) {
    return course;
  }

  return {
    ...course,
    lessons: course.lessons.map((lesson, index) => (index === lessonIndex ? nextLesson : lesson)),
  };
};

const updateBlockInLesson = (
  course: Course,
  lessonId: string,
  blockId: string,
  updater: (block: CourseBlock) => CourseBlock,
): Course =>
  updateLesson(course, lessonId, (lesson) => {
    const blockIndex = lesson.blocks.findIndex((block) => block.id === blockId);

    if (blockIndex === -1) {
      return lesson;
    }

    return {
      ...lesson,
      blocks: lesson.blocks.map((block, index) => (index === blockIndex ? updater(block) : block)),
    };
  });

const cloneBlockDefaults = (family: BlockFamily): Omit<CourseBlock, "id"> => {
  const defaults = blockDefaults[family];

  return {
    ...defaults,
    content: cloneValue(defaults.content),
    settings: cloneValue(defaults.settings),
  };
};

const cloneRecord = <T extends Record<string, unknown>>(value: T): T => cloneValue(value);

const cloneValue = <T>(value: T): T => {
  if (typeof structuredClone === "function") {
    return structuredClone(value);
  }

  return JSON.parse(JSON.stringify(value)) as T;
};

const clampInsertionIndex = (index: number | undefined, length: number): number => {
  if (index === undefined || Number.isNaN(index)) {
    return length;
  }

  return Math.max(0, Math.min(Math.trunc(index), length));
};

const insertItem = <T>(items: T[], index: number, item: T): T[] => {
  const nextItems = [...items];
  nextItems.splice(index, 0, item);

  return nextItems;
};

const nextId = (course: Course, prefix: "lesson" | "block"): string => {
  const ids = new Set([
    course.id,
    ...course.lessons.map((lesson) => lesson.id),
    ...course.lessons.flatMap((lesson) => lesson.blocks.map((block) => block.id)),
  ]);
  let index = ids.size + 1;
  let id = `${prefix}-${index}`;

  while (ids.has(id)) {
    index += 1;
    id = `${prefix}-${index}`;
  }

  return id;
};
